```bash
< PROJECT ROOT >
   |
   |-- config/                              # Реализует конфигурацию приложения
   |    |-- settings.py                    # Определяет Глобальные Настройки
   |    |-- wsgi.py                        # Запускает приложение в рабочей среде
   |    |-- urls.py                        # Определение URL-адресов, обслуживаемых всеми приложениями /узлами
   |
   |-- apps/                               # Каталог для всех приложений
   |    |
   |    |-- core/                          # Главное приложение
   |    |    |-- services/                 # Бизнес логика 
   |    |    |-- views.py                  # Контроллеры
   |    |    |-- urls.py                   # Определение URL в данном приложении  
   |    |
   |    |-- authentication/                # Приложение аутентификации 
   |    |    |-- urls.py                   # URL для аутентификации
   |    |    |-- views.py                  # Контроллеры аутентификации
   |    |    |-- forms.py                  # Формы для входа и регистрации 
   |    |
   |    |-- static/
   |    |    |-- <css, JS, images>         # CSS файлы, Java Scripts файлы
   |    |
   |    |-- templates/                     # Шаблоны для отображения в контроллерах
   |         |-- includes/                 # Фрагменты и компоненты HTML
   |         |    |-- navigation.html      # Компонент верхнего меню
   |         |    |-- sidebar.html         # Компонент бокового меню
   |         |    |-- footer.html          # Компонент Footer
   |         |    |-- scripts.html         # Скрипты, используемые во всех шаблонах
   |         |
   |         |-- layouts/                   # Мастер шаблоны
   |         |    |-- base-fullscreen.html  # Используется для шаблонов аутентификации
   |         |    |-- base.html             # Используется для основных шаблонов
   |         |
   |         |-- accounts/                  # Шаблоны аутентификации
   |         |    |-- login.html            # Шаблон входа
   |         |    |-- register.html         # Шаблон регистрации
   |         |
   |         |-- core/                      # Шаблоны приложения
   |              |-- index.html            # Шаблон index
   |              |-- 404-page.html         # Шаблон 404
   |              |-- *.html                # Прочие шаблоны
   |
   |-- requirements/                        # Модули окружения
   |
   |-- .env                                 # Переменные окружения
   |-- manage.py                            # Запуск приложения / стандартные скрипты Django
   |
   |-- ************************************************************************
```
